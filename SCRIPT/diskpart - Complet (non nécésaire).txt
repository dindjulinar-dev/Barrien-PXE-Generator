select disk 0
clean
convert gpt

rem --- 1. Partition Système EFI ---
create partition efi size=100
format fs=fat32 quick label="System"
assign letter=S

rem --- 2. Partition MSR (Réservé Microsoft) ---
create partition msr size=16

rem --- 3. Partition Temporaire Locale (15 Go) ---
create partition primary size=15360
format fs=ntfs quick label="LOCAL_TEMP"
assign letter=L

rem --- 4. Partition Principale Windows (Tout l'espace restant) ---
create partition primary
format fs=ntfs quick label="Windows"
assign letter=C

rem --- 5. Partition de Récupération (WinRE) ---
shrink minimum=1024
create partition primary
format fs=ntfs quick label="Recovery"
set id="de94bba4-06d1-4d40-a16a-bfd50179d6ac"
gpt attributes=0x8000000000000001
assign letter=R
